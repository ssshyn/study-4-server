
## 1. Docker Compose가 필요한 이유

```
실제 서비스는 컨테이너 하나로 동작하지 않음.

Spring Boot 앱 하나를 실행하려면:
	- Spring Boot 앱 컨테이너
	- MySQL 컨테이너
	- Redis 컨테이너
	- Kafka 컨테이너
	  
docker run 명령어를 4번 수동으로 실행하고,
네트워크 연결하고, 의존 순서 맞추고 등 번거롭다.

Docker Compose: 여러 컨테이너를 하나의 YAML 파일로 정의하고 한 명령어로 실행/종료
```

## 2. docker-compose.yml 기본 구조

```yaml
version: "3.9"

services:                                     # 서비스 이름(컨테이너명)
	app:
		build:
			context: .                        # Dockerfile 위치
			dockerfile: Dockerfile
		image: myapp:latest                   # 또는 빌드 대신 기존 이미지 사용
		container_name: myapp
		ports:
			- "8080:8080"			          # 호스트:컨테이너
		environment:
			SPRING_PROFILES_ACTIVE: loca
			DB_URL: jdbc:mysql://db:3306/mydb
			DB_USERNAME: root
			DB_PASSWORD: secret
		depends_on:
			db:
				condition: service_healthy    # DB가 healthy 상태일 때 시작
			redis:
				condition: service_started
		networks:
			- backend
		volumes:
			- app-logs:/app/logs
		restart: unless-stopped     # 컨테이너 재시작 정책
		
db:
	image: mysql:8.0
	container_name: mysql
	environment:
		MYSQL_ROOT_PASSWORD: secret
		MYSQL_DATABASE: mydb
		MYSQL_USER: appuser
		MYSQL_PASSWORD: apppass
	ports:
		- "3306:3306"      # 로컬 접속용 (운영 x)
	volumes:
		- mysql-data:/var/lib/mysql  # 데이터 영속성
		- ./init.sql:/docker-entrypoint-initdb.d/init.sql # 초기 SQL
	healthcheck:
		test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
		interval: 10s
		timeout: 5s
		retries: 5
	networks:
		- backend
		  
redis:
	image: redis:7.2-alpine
	container_name: redis
	command: redis-server --requirepass redispass
	ports:
		- "6379:6379"
	volumes:
		- redis-data:/data
	networks:
		- backend
		  
networks:
	backend:
		driver: bridge    # 기본값

volumes:
	mysql-data:
	redis-data:
	app-logs:
```

## 3. 주요 Compose 명령어

```bash
# 모든 서비스 시작 (백그라운드)
docker compose up -d

# 특정 서비스만 시작
docker compose up -d db redis

# 빌드 후 시작 (코드 변경 시)
docker compose up -d --build app

# 로그 확인
docker compose logs -f
docker compose logs -f app   # 특정 서비스

# 서비스 목록 및 상태
docker compose ps

# 특정 서비스 재시작
docker compose restart app

# 서비스 중지 (컨테이너 유지)
docker compose stop

# 서비스 중지 + 컨테이너 삭제
docker compose down

# 서비스 중지 + 컨테이너 + 볼륨까지 삭제 (초기화)
docker compose down -v

# 특정 서비스 컨테이너 내부 접속
docker compose exec app sh
docker compose exec db mysql -uroot -psecret mydb
```

## 4. 환경별 Compose 파일 분리

### 4-1. 오버라이드 패턴

```yaml
# docker-compose.yml (공통 기반)
services:
	app:
		image: myapp
		environment:
			JAVA_OPTS: "-Xmx512m"
	db:
		image: mysql:8.0
		environment:
			MYSQL_ROOT_PASSWORD: ${DB_PASSWORD}
```

```yaml
# docker-compose.override.yml (로컬 개발용 - 자동으로 병합)
services:
	app:
		build: .      # 로컬에서는 빌드
		volumes:
			- ./src:/app/src   # 소스 코드 실시간 반영
		environment:
			SPRING_PROFILES_ACTIVE: local
			DEBUG: "true"
	db:
		ports:
			- "3306:3306"     # 로컬에서는 DB 포트 오픈
```

```yaml
# docker-compose.prod.yml (운영용)
services:
	app:
		image: 123456789.dkr.ecr.ap-northeast-2.amazonaws.com/myapp:${VERSION}
		restart: always
		deploy:
			resources:
				limits:
					memory: 1g
					cpus: "2.0"
	db:
		# ports 없음 - 운영에서 DB 포트 외부 노출 금지
		environment:
			MYSQL_ROOT_PASSWORD: ${DB_PASSWORD}
```

```bash
# 로컬 개발 (override 자동 적용)
docker compose up -d

# 운영 배포
docker compose -f docker-compose.yml -f docker-compose.prod.yml up -d
```

## 5. 실제 개발 환경 구성 예씨
```yaml
# docker-compose.yml — 백엔드 개발 환경 전체
version: "3.9"

services:
  # 앱은 로컬에서 실행, 인프라만 컨테이너로
  mysql:
    image: mysql:8.0
    environment:
      MYSQL_ROOT_PASSWORD: local
      MYSQL_DATABASE: myapp_dev
    ports:
      - "3306:3306"
    volumes:
      - mysql-data:/var/lib/mysql
      - ./scripts/init.sql:/docker-entrypoint-initdb.d/01-init.sql
    command: --character-set-server=utf8mb4 --collation-server=utf8mb4_unicode_ci

  redis:
    image: redis:7.2-alpine
    ports:
      - "6379:6379"
    command: redis-server --maxmemory 256mb --maxmemory-policy allkeys-lru

  kafka:
    image: confluentinc/cp-kafka:7.5.0
    ports:
      - "9092:9092"
    environment:
      KAFKA_NODE_ID: 1
      KAFKA_PROCESS_ROLES: broker,controller
      KAFKA_LISTENERS: PLAINTEXT://0.0.0.0:9092,CONTROLLER://0.0.0.0:9093
      KAFKA_ADVERTISED_LISTENERS: PLAINTEXT://localhost:9092
      KAFKA_CONTROLLER_QUORUM_VOTERS: 1@kafka:9093
      KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
      CLUSTER_ID: MkU3OEVBNTcwNTJENDM2Qk

  kafka-ui:
    image: provectuslabs/kafka-ui:latest
    ports:
      - "8989:8080"
    environment:
      KAFKA_CLUSTERS_0_NAME: local
      KAFKA_CLUSTERS_0_BOOTSTRAPSERVERS: kafka:9092
    depends_on:
      - kafka

  elasticsearch:
    image: elasticsearch:8.11.0
    environment:
      discovery.type: single-node
      xpack.security.enabled: false
      ES_JAVA_OPTS: "-Xms512m -Xmx512m"
    ports:
      - "9200:9200"
    volumes:
      - es-data:/usr/share/elasticsearch/data

volumes:
  mysql-data:
  es-data:
```

    password: local

## 6. Heathcheck와 depends_on

```yaml
services:
	app:
		depends_on:
			db:
				condition: service_healthy   # DB가 healthy 상태여야 시작
			redis:
				condition: service_started   # Redis가 시작만 되면 OK
	db:
		heathcheck:
			# MySQL이 실제로 쿼리를 받을 수 있는지 확인
			test: ["CMD", "mysqladmin", "ping", "-h", "localhost", "-u", "root", "-p${MYSQL_ROOT_PASSWORD}"]
			interval: 10s  # 10초마다 체크
			timeout: 5s    # 5초 안에 응답 없으면 실패
			retries: 5     # 5번 실패하면 unhealthy
			start_period: 30s  # 최초 30초는 실패해도 카운트 안 함
```

## 7. Compose 네트워크

```yaml
services:
	app:
		networks:
			- frontend  # nginx와 통신
			- backend   # db, redis와 통신
			
	nginx:
		networks:
			- frontend  # 외부 트래픽 수신
	
	db:
		networks:
			- backend   # app만 접근 가능, nginx는 접근 불가
			  
networks:
	frontend:
	backend:
		internal: true   # 외부 인터넷 접근 차단
```

```
서비스 간 통신:
	컨테이너 내에서 다른 서비스에 접근할 때 서비스명을 호스트명으로 사용
	
	app 컨테이너에서 DB 접근:
		jdbc:mysql://db:3306/mydb    ← 서비스명 "db"를 호스트명으로
		
	app 컨테이너에서 Redis 접근:
		host: redis, port: 6379      ← 서비스명 "redis"를 호스트명으로
```
